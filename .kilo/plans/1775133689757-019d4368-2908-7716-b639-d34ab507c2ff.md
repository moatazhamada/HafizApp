# Home Screen Widget - Daily Ayah Inspiration

## Plan Overview

Add a user-opt-in home-screen widget displaying a randomly chosen Ayah or partial Ayah as daily inspiration. This feature requires platform-specific implementations for Android (App Widget) and iOS (WidgetKit).

---

## 1. Data Architecture

### 1.1 Ayah Corpus Storage Options

| Option | Pros | Cons |
|--------|------|------|
| **Local JSON (Recommended)** | No API calls, works offline, full control | Larger app size (~2MB for all Ayah) |
| **Room Database** | Queryable, efficient | More complex setup |
| **Public API** | Smaller initial app size | Requires internet, dependency on external service |

**Recommendation:** Bundle selected Ayah in local JSON (~500 key Ayah for variety) - sufficient for daily inspiration without internet dependency.

func getTimeline(in context: Context, completion: @escaping (Timeline<Entry>) -> Void) {
  let currentDate = Date()
  let midnight = Calendar.current.startOfDay(for: currentDate)
  
  // Entry for now + entries for next 7 days at midnight
  let entries = (0..<7).map { dayOffset -> AyahEntry in
    let entryDate = Calendar.current.date(byAdding: .day, value: dayOffset, to: midnight)!
    return AyahEntry(date: entryDate, ayah: randomAyah())

  completion(Timeline(entries: entries, policy: .atEnd))
}

### 7.3 Random Selection Use Case

```dart
// lib/domain/usecase/get_random_ayah.dart
import 'package:hafiz_app/core/utils/widget_preferences.dart';
import 'package:hafiz_app/data/model/ayah_model.dart';
import 'package:hafiz_app/data/repository/ayah_repository.dart';
import 'package:hafiz_app/injection_container.dart';
import 'package:shared_preferences/shared_preferences.dart';

class GetRandomAyah {
  final AyahRepository _repository = sl<AyahRepository>();
  final SharedPreferences _prefs = sl<SharedPreferences>();
  
  Future<AyahModel> call({bool avoidRecent = true}) async {
    final allAyah = await _repository.getAllAyah();
    
    // Get recent Ayah to avoid repetition
    List<int> recentIds = [];
    if (avoidRecent) {
      final recentJson = _prefs.getString(WidgetPreferences.KEY_WIDGET_RECENT_AYAH);
      if (recentJson != null) {
        recentIds = recentJson.split(',').map((e) => int.tryParse(e) ?? -1).toList();
      }
    }
    
    // Filter out recent
    final candidates = allAyah.where((a) => !recentIds.contains(a.chapter * 1000 + a.verse)).toList();
    
    // Weighted random selection
    final weighted = candidates.map((a) => List.filled(a.priority, a)).expand((e) => e).toList();
    final selected = weighted[DateTime.now().millisecondsSinceEpoch % weighted.length];
    
    // Update recent list (keep last 10)
    _updateRecentList(selected.chapter, selected.verse);
    
    return selected;
  }
  
  void _updateRecentList(int chapter, int verse) {
    final key = chapter * 1000 + verse;
    final recentJson = _prefs.getString(WidgetPreferences.KEY_WIDGET_RECENT_AYAH) ?? '';
    final recent = recentJson.isEmpty 
      ? <int>[]
      : recentJson.split(',').map((e) => int.tryParse(e) ?? -1).toList();
    
    recent.add(key);
    if (recent.length > 10) recent.removeAt(0);
    
    _prefs.setString(WidgetPreferences.KEY_WIDGET_RECENT_AYAH, recent.join(','));
  }
}
```

### 7.4 Android Widget Provider

```kotlin
// android/app/src/main/kotlin/com/hafiz/app/hafiz_app/HafizWidgetProvider.kt
package com.hafiz.app.hafiz_app

import android.appwidget.AppWidgetManager
import android.appwidget.AppWidgetProvider
import android.content.Context
import android.content.SharedPreferences
import android.widget.RemoteViews
import es.antonborri.home_widget.HomeWidgetPlugin

class HafizWidgetProvider : AppWidgetProvider() {
    
    override fun onUpdate(
        context: Context,
        appWidgetManager: AppWidgetManager,
        appWidgetIds: IntArray
    ) {
        for (appWidgetId in appWidgetIds) {
            val widgetData = HomeWidgetPlugin.getData(context)
            val views = RemoteViews(context.packageName, R.layout.hafiz_widget)
            
            // Get Ayah data from shared preferences
            val arabicText = widgetData.getString("widget_arabic", "﴿ بِسْمِ اللَّهِ الرَّحْمَنِ الرَّحِيمِ ﴾")
            val translation = widgetData.getString("widget_translation", "In the name of Allah, the Most Gracious, the Most Merciful")
            val reference = widgetData.getString("widget_reference", "Surah Al-Fatiha: 1")
            
            views.setTextViewText(R.id.widget_arabic, arabicText)
            views.setTextViewText(R.id.widget_translation, translation)
            views.setTextViewText(R.id.widget_reference, reference)
            
            // Set click to open app
            val intent = Intent(context, MainActivity::class.java)
            val pendingIntent = PendingIntent.getActivity(
                context, 0, intent,
                PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
            )
            views.setOnClickPendingIntent(R.id.widget_container, pendingIntent)
            
            appWidgetManager.updateAppWidget(appWidgetId, views)
        }
    }
}
```

// ios/Runner/AyahWidget.swift
import WidgetKit
import SwiftUI

struct AyahEntry: TimelineEntry {
    let date: Date
    let arabic: String
    let translation: String
    let reference: String
}

struct AyahProvider: TimelineProvider {
    func placeholder(in context: Context) -> AyahEntry {
        AyahEntry(date: Date(), arabic: "﴿ بِسْمِ اللَّهِ ﴾", translation: "In the name of Allah", reference: "Surah Al-Fatiha:1")
    }
    
    func getSnapshot(in context: Context, completion: @escaping (AyahEntry) -> Void) {
        let entry = AyahEntry(date: Date(), arabic: "﴿ بِسْمِ اللَّهِ الرَّحْمَنِ الرَّحِيمِ ﴾", translation: "In the name of Allah, the Most Gracious, the Most Merciful", reference: "Surah Al-Fatiha:1")
        completion(entry)
    }
    
    func getTimeline(in context: Context, completion: @escaping (Timeline<AyahEntry>) -> Void) {
        let currentDate = Date()
        let midnight = Calendar.current.startOfDay(for: currentDate)
        
        var entries: [AyahEntry] = []
        for dayOffset in 0..<7 {
            let entryDate = Calendar.current.date(byAdding: .day, value: dayOffset, to: midnight)!
            let ayah = getRandomAyah() // Call to shared Flutter method
            let entry = AyahEntry(date: entryDate, arabic: ayah.arabic, translation: ayah.translation, reference: ayah.reference)
            entries.append(entry)
        }
        
        let timeline = Timeline(entries: entries, policy: .atEnd)
        completion(timeline)
    }
}

struct AyahWidgetEntryView: View {
    var entry: AyahProvider.Entry
    
    var body: some View {
        VStack(alignment: .leading, spacing: 8) {
            Text(entry.arabic)
                .font(.system(size: 18, weight: .medium))
                .foregroundColor(.primary)
                .lineLimit(4)
                .multilineTextAlignment(.leading)
            
            Text(entry.translation)
                .font(.system(size: 12))
                .foregroundColor(.secondary)
                .lineLimit(2)
            
            Spacer()
            
            Text(entry.reference)
                .font(.system(size: 10))
                .foregroundColor(.secondary)
        }
        .padding()
    }
}

@main
struct AyahWidget: Widget {
    let kind: String = "AyahWidget"
    
    var body: some WidgetConfiguration {
        StaticConfiguration(kind: kind, provider: AyahProvider()) { entry in
            AyahWidgetEntryView(entry: entry)
        }
        .configurationDisplayName("Daily Ayah")
        .description("Receive daily inspiration from the Quran")
        .supportedFamilies([.systemSmall, .systemMedium, .systemLarge])
    }
}

class WidgetManager {

      androidName: 'HafizWidgetProvider',

    );
  }
